AngularJS使用过程中的Exception

1:向后台传参太多会出现Exception:HTTP 413error!

  //angularjs 方式，参数实体比较大时，会出现HTTP 413错误
  $resource(roleRes.saveJsonUrl,{roleResData:returnData,rolekey:$scope.selectObj.rolekey,resource_type_key:resourceTypeKey},actions).save(function(retData){
	  alert(retData.message);   
	});
	//解决方案：ajax 方式  
	$.ajax({
		url:'${appName}/t00_role_resource/saveJsonData.ajax',
		data:{roleResData:returnData,rolekey:$scope.selectObj.rolekey,resource_type_key:resourceTypeKey},
		dataType:'json',
		type:'post',
		success:function(retData){
		  alert(retData.message); 
		}
	});
